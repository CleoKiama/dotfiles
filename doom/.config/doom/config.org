* Evil Mode
** Escape Sequence
Map =jj= to Escape in insert mode via =evil-escape=.
The 0.2s delay is intentional — too short and typing "jump" triggers it accidentally.
#+BEGIN_SRC emacs-lisp
(after! evil-escape
  (setq evil-escape-key-sequence "jj"
        evil-escape-delay 0.2
        evil-escape-unordered-key-sequence t)
  (evil-escape-mode 1))

;; Fix: org-agenda 'j' should go to next line, not open calendar

(after! org-agenda
  (define-key org-agenda-mode-map (kbd "j") #'org-agenda-next-line)
  (define-key org-agenda-mode-map (kbd "k") #'org-agenda-previous-line))

** Activation Timestamp
Automatically stamp a property =ACTIVATED= when a task moves to =NEXT=,
so you can track how long something has been your active focus.
#+BEGIN_SRC emacs-lisp
  (defun log-todo-state-change (&rest args)
    "Log timestamps for various todo states."
    (let ((state (org-get-todo-state)))
      (cond
       ;; When moving to NEXT, log ACTIVATED
       ((and (string= state "NEXT")
             (not (org-entry-get nil "ACTIVATED")))
        (org-entry-put nil "ACTIVATED" (format-time-string "[%Y-%m-%d]")))

       ;; When @waiting tag is added, log WAITING_SINCE
       ((and (member "@waiting" tags)
           (not (org-entry-get nil "WAITING_SINCE")))
       (org-entry-put nil "WAITING_SINCE" (format-time-string "[%Y-%m-%d]")))

        ;; When @waiting tag is removed, clear WAITING_SINCE
       ((and (org-entry-get nil "WAITING_SINCE")
           (not (member "@waiting" tags)))
         (org-entry-put nil "WAITING_SINCE" nil)))))

  (add-hook 'org-after-todo-state-change-hook #'log-todo-state-change)
  (add-hook 'org-after-tag-change-hook #'log-todo-state-change) ;; Also trigger on tag changes!

  ;; Auto-add CREATED property on capture (via template handles this)
  ;; But also ensure any new TODO gets a CREATED timestamp
  (defun log-task-creation-time ()
    "Log CREATED timestamp when a new TODO is created."
    (when (and (string= (org-get-todo-state) "TODO")
               (not (org-entry-get nil "CREATED")))
      (org-entry-put nil "CREATED" (format-time-string "[%Y-%m-%d]"))))

  (add-hook 'org-after-todo-state-change-hook #'log-task-creation-time)
#+END_SRC

* Linear.app
Integration with Linear.app for issue tracking.
#+BEGIN_SRC emacs-lisp
(use-package! linear-emacs
  :after org
  :commands (linear-emacs-list-issues
             linear-emacs-new-issue
             linear-emacs-enable-org-sync
             linear-emacs-sync-org-to-linear)
  :init
  (setq linear-emacs-org-file-path (expand-file-name "linear.org" my/org-gtd))

  :config
  ;; API Key loading logic
  (defun my/linear-load-api-key-from-auth-source ()
    "Load Linear API key from auth-source (machine api.linear.app login apikey)."
    (require 'auth-source)
    (let* ((auth-info (auth-source-search :host "api.linear.app" :user "apikey" :max 1))
           (secret (when auth-info
                     (funcall (plist-get (car auth-info) :secret)))))
      (if secret
          (setq linear-emacs-api-key secret)
        (linear-emacs-load-api-key-from-env))))

  (my/linear-load-api-key-from-auth-source)

  ;; State mapping to match our org-todo-keywords
  (setq linear-emacs-issues-state-mapping
        '(("Backlog"    . "TODO")
          ("Todo"       . "TODO")
          ("In Progress" . "WIP")
          ("In Review"   . "REVIEW")
          ("Blocked"    . "HOLD")
          ("Done"       . "DONE")))

  ;; Priority mapping
  (setq linear-emacs-org-priority-mapping
        '((0 . nil)   ; No priority
          (1 . "A")   ; Urgent
          (2 . "B")   ; High
          (3 . "C")   ; Medium
          (4 . "D"))) ; Low

  ;; Auto-enable sync for linear.org
  (defun my/enable-linear-org-sync ()
    "Enable Linear-org synchronization when linear.org is opened."
    (when (and buffer-file-name
               (string-match-p "linear\\.org$" buffer-file-name))
      (linear-emacs-enable-org-sync)
      (message "Linear-org synchronization enabled for this buffer")))

  (add-hook 'find-file-hook #'my/enable-linear-org-sync)

  ;; Auto-sync state changes to Linear
  (add-hook 'org-after-todo-state-change-hook
            (lambda ()
              (when (and buffer-file-name
                         (string-match-p "linear\\.org$" buffer-file-name)
                         (fboundp 'linear-emacs-sync-org-to-linear))
                (linear-emacs-sync-org-to-linear))))

  ;; Keybindings
  (map! :leader
        (:prefix-map ("l" . "linear")
         :desc "List Linear issues" "l" #'linear-emacs-list-issues
         :desc "Create new issue" "n" #'linear-emacs-new-issue
         :desc "Sync current issue" "s" #'linear-emacs-sync-org-to-linear
         :desc "Enable org sync" "e" #'linear-emacs-enable-org-sync
         :desc "Disable org sync" "d" #'linear-emacs-disable-org-sync
         :desc "Test connection" "t" #'linear-emacs-test-connection)))
#+END_SRC
