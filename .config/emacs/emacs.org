#+begin_src emacs-lisp
(use-package projectile
  :custom
  (projectile-indexing-method 'hybrid)       ;; hybrid: external tool + Elisp-level ignores
  (projectile-require-project-root 'prompt)   
  (projectile-project-search-path
   '("/data/code/" ("/data/org/" . 1) "/media/DevDrive/thegoodco" "/media/DevDrive" "/home/cleo/dotfiles"))
  (projectile-switch-project-action #'projectile-find-file)
  (projectile-auto-cleanup-known-projects t)
  (projectile-git-submodule-command nil)       ;; disable submodule scanning (perf)

  :config
  ;; 1. Add ignored directories to Projectile's native ignore list
  (dolist (dir cl/ignored-directories)
    (add-to-list 'projectile-globally-ignored-directories dir))

  ;; 2. Lambda-based generic command (Doom-inspired).
  ;;    Re-evaluates at RUNTIME, so it always finds fd/rg regardless of
  ;;    when exec-path-from-shell runs.  This is the key fix for the freeze.
  (setq projectile-generic-command
        (lambda (_)
          (cond
           ((executable-find "fd")
            (concat "fd . -0 --type f --hidden --follow --exclude .git "
                    (mapconcat (lambda (dir) (concat "--exclude " dir))
                               cl/ignored-directories " ")))
           ((executable-find "rg")
            (concat "rg --files -0 --hidden --follow --color=never --no-require-git "
                    (mapconcat (lambda (dir) (format "-g '!%s'" dir))
                               cl/ignored-directories " ")))
           (t "find . -type f -print0"))))

  ;; 3. Force projectile to ALWAYS use the generic command above,
  ;;    never VCS-specific commands (git ls-files, hg, etc.).
  (advice-add
   #'projectile-get-ext-command :around
   (lambda (fn &rest args)
     (if (functionp projectile-generic-command)
         (funcall projectile-generic-command (car args))
       (apply fn args))))

  ;; project.el integration
  (add-hook 'project-find-functions #'project-projectile)

  ;; Leader keys
  (cl/leader-keys
    "p"  '(:ignore t :which-key "projects")
    "pf" '(projectile-find-file :which-key "find file")
    "pw" '(consult-ripgrep :which-key "ripgrep")
    "ps" '(projectile-switch-project :which-key "switch project")
    "pp" '(projectile-run-project :which-key "run project")
    "pt" '(projectile-test-project :which-key "test project")
    "pd" '(projectile-dired :which-key "project dired")))

(projectile-mode +1)

** Tangle this file on save

Guarded so saving *this specific file* re-tangles =emacs.el=, but
saving any other Org file on the system doesn't. This keeps the
tangled file fresh for git diffs / anyone inspecting it directly,
even though =init.el= itself doesn't strictly need it thanks to
~org-babel-load-file~'s own mtime check at startup.

#+begin_src emacs-lisp
(defun cl/org-babel-tangle-config ()
  (when (string-equal (file-truename buffer-file-name)
                       (file-truename (expand-file-name "emacs.org" user-emacs-directory)))
    (let ((org-confirm-babel-evaluate nil))
      (org-babel-tangle))))

(add-hook 'org-mode-hook
          (lambda ()
            (add-hook 'after-save-hook #'cl/org-babel-tangle-config nil 'local)))
#+end_src
